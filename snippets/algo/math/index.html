\[\begin{eqnarray} \\A = Km + n \leqq & i = K*j & \leqq B = Km' + n' & <=> \\m + n/K \leqq & j & \leqq m' + n'/K \\ \\以下を考慮 \\0 \leqq & n/K, n'/K & < 1 \\ \\Kの倍数から整数に帰着 \\m \leqq & j & \leqq m' & (n/K = 0) \\m < & j & \leqq m' & (else) \\ \end{eqnarray}\]
