#include <iostream>
#include <string>
using namespace std;
void hello(string h = "Ben", string s = "hello, world") {
  cout << h << " says " << s << endl;
}
int main() { hello(); }

  cout << h << " says " << s << endl;

int main() { hello("John"); }
