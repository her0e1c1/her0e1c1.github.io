kwargs.scm¶

デフォルト引数はもっとも右側に指定していく

#include <iostream>
#include <string>
using namespace std;
void hello(string h = "Ben", string s = "hello, world") {
  cout << h << " says " << s << endl;
}
int main() { hello(); }
