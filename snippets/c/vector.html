indexAt¶

Refer nth element.

Using a vector, you can access an element of it for a constant time.

// IMPLEMENT: vectorの実装(indexAt)
#include <stdio.h>
#include <stdlib.h>
#include <vector.h>

int indexAt(vector_t *head, int index) {
  if (0 <= index && index < head->size)
    return head->data[index];

  exit(1);
}

insertAt¶

You need to move an element one by one from index. So complexity is O(n)

// IMPLEMENT: insertAt(vector)
#include <stdio.h>
#include <stdlib.h>
#include <vector.h>

void insertAt(vector_t *head, int index, int value) {
  int *data = realloc(head->data, sizeof(int) * (head->size + 1));
  if (data == NULL)
    exit(1);

  head->data = data;
  int i = head->size;
  while (i > index) {
    head->data[i] = head->data[i - 1];
    i--;
  }
  head->data[index] = value;
  head->size++;
}

deleteAt¶

// IMPLEMENT: deleteAt(vector)
#include <stdio.h>
#include <stdlib.h>
#include <vector.h>

// 指定したindex以降の順番を一つずつ手前にずらす必要があるため、O(n)
void deleteAt(vector_t *head, int index) {
  if (index == 0)
    return;

  int i = index;
  while (i < head->size - 1) {
    head->data[i] = head->data[i + 1];
    i++;
  }
  realloc(head->data, sizeof(int) * (head->size - 1));
  // head->data以外はmallocで確保していないのでfreeしてはいけない
  // free(head->data + (head->size - 1));
  head->size--;
}
