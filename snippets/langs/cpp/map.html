insert¶

$ cpe 'map<string, int>m={{"a", 1}}; m.insert(map<string,int>::value_type("b", 2)); P(m["b"])'
m["b"] = 2

$ cpe 'map<string, int>m={{"a", 1}}; m.insert(make_pair("b", 2)); P(m["b"])'
m["b"] = 2

update¶

$ cpe 'map<string, int>m={{"a", 1}}; m["a"]=2; P(m["a"])'
m["a"] = 2
